algoritmo "Soma_10_Numeros"
var
   num, soma, contador: inteiro
inicio

   soma <- 0
   contador <- 1

   enquanto contador <= 10 faca
      escreva("Digite o ", contador, "º número: ")
      leia(num)

      soma <- soma + num
      contador <- contador + 1
   fimenquanto

   escreval("A soma total é: ", soma)

fimalgoritmo
